#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

#define ll long long
int main() {
    int kase;
    cin>>kase;
    while(kase--) {
        int n;
        cin>>n;
        vector<vector<int>> v(n, vector<int>(n));
        for(int i=0; i<n; i++)
            for(int j=0; j<n; j++)
                cin>>v[i][j];
        vector<int> destroy_queue(n);
        for(int i=0; i<n; i++)
            cin>>destroy_queue[i];
        reverse(destroy_queue.begin(), destroy_queue.end());

        vector<int> g;
        ll sum = 0;
        for(int k=0; k<n; k++){
            int x = destroy_queue[k];
            g.push_back(x);
            for(int i=0; i<n; i++){
                for(int j=0; j<n; j++){
                    if(i==j) continue;
                    if(v[i][j] > v[i][x] + v[x][j])
                        v[i][j] = v[i][x] + v[x][j];
                }
            }
            for(int i=0; i<g.size(); i++){
                for(int j=0; j<g.size(); j++){
                    if(i==j) continue;
                    sum += v[g[i]][g[j]];
                }
            }
        }
        // for(int i=0; i<n; i++, cout<<endl)
        //     for(int j=0; j<n; j++)
        //         cout << v[i][j] << " ";
        cout << sum << endl;
    }

    return 0;
}
